              #include < stdio.h >

              int main() {
                  int n, i, sum = 0;
                  float mean;
              
                  printf("Enter the length of the vector: ");
                  scanf("%d", &n);
              
                  int vec[n]; // create vector with length n
              
                  printf("Enter %d integers for the vector: \n", n);
                  for (i = 0; i < n; i++) {
                      scanf("%d", &vec[i]); // read input and store in vector
                      sum += vec[i]; // accumulate sum of vector elements
                  }
              
                  mean = (float) sum / n; // calculate arithmetic mean
              
                  printf("The arithmetic mean of the vector is: %f\n", mean);
              
                  return 0;
              }
